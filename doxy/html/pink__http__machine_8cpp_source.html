204 pink_http_machine::HTTP_CODE pink_http_machine::parse_request_line(char *text){
205  // 解析 URL
206  // strpbrk(): string pointer break
207  // 找到任意字符出现的第一个位置
208  url = strpbrk(text, " \t");
209  if(!url){
210  return BAD_REQUEST;
211  }
212  *url++ = '\0';
213  // strspn(): Get span of character set in string
214  // 返回只包含字符集字符的前缀串长度
215  url += strspn(url, " \t");
216  if(strncasecmp(url, "http://", 7) == 0){
217  url += 7;
218  // strchr(): locate the first occurence of character in string
219  url = strchr(url, '/');
220  }
221  if(!url || url[0] != '/'){
222  return BAD_REQUEST;
223  }
224 
225  // ==================================================================_
226  // 解析请求方法
227  method = parse_request_method(text);
228  if(method == UNKNOWN_METHOD){

230  return BAD_REQUEST;
231  }

234  // 解析 HTTP 版本
235  version = strpbrk(url, " \t");
236  if(!version){
237  return BAD_REQUEST;
238  }

243  return BAD_REQUEST;

248  // 尚未完成全部解析

349 pink_http_machine::HTTP_CODE pink_http_machine::do_request(){
350

352  int len = strlen(conf.doc_root);
353  int url_len = strlen(url);
354 
355  if(url_len > FILENAME_LEN - len - 1){
356  return BAD_REQUEST;
357  }

359  request_file[len + url_len] = '\0';
360

362  if(stat(request_file, &file_stat) < 0){
363  return NO_RESOURCE;
364  }
365 
366  // S_IROTH：S_ Inode Read Other，其他用户具有可读取权限
367  if(!(file_stat.st_mode & S_IROTH)){
368  // 无读取权限
369  return FORBIDDEN_REQUEST;
370  }
371 
372  // 是文件夹返回1，不是返回0
373  if(S_ISDIR(file_stat.st_mode)){
374  return BAD_REQUEST;
375  }
376 
377  int fd = open(request_file, O_RDONLY);
378  // 第一个 NULL 表示让系统自动选定 start 地址
379  // PROT_READ： 映射区域可被读取
380  // MAP_PRIVATE： 私有的进行map，进行的是写时复制，
381  // 本进程对映射区域的改变对硬盘上的文件以及其他映射这个文件的进程来说是不可见的。
382  // 最后一个参数：offset = 0
383  file_address = (char*)mmap(NULL, file_stat.st_size, PROT_READ,
